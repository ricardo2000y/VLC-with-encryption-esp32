114static bool process_buffer_recursive(char* output_str, char* hex_str, size_t max_length, size_t* current_length, size_t* hex_length) {
115 uint32_t value;
116 if (ringBufferPop(ring_buffer_RX, &value)) {
117 value ^= key_generator(&RX_encryption_vars);
118 unsigned char b0, b1, b2, b3;
119 splitUint32ToChars(value, &b0, &b1, &b2, &b3);
120
121 // Append bytes to the output string
122 if (*current_length < max_length - 1) output_str[(*current_length)++] = b0;
123 if (*current_length < max_length - 1) output_str[(*current_length)++] = b1;
124 if (*current_length < max_length - 1) output_str[(*current_length)++] = b2;
125 if (*current_length < max_length - 1) output_str[(*current_length)++] = b3;
126
127 // Append hex representation
128 *hex_length += snprintf(hex_str + *hex_length, max_length - *hex_length,
129 "%02X%02X%02X%02X ", b0, b1, b2, b3);
130
131 // Recursively process the next value
132 return process_buffer_recursive(output_str, hex_str, max_length, current_length, hex_length);
133 }
134 return false; // Ring buffer is empty
135}

140static void process_reception_complete(void) {
141 char output_str[BUFFER_MAX_SIZE*4 + 1];
142 char hex_str[BUFFER_MAX_SIZE*9 + 1]; // 8 chars per uint32 (2 per byte) + space

144 size_t hex_length = 0;

159 ESP_LOGI(RX_TAG, "Received data contains non-printable characters");

198static bool IRAM_ATTR timer_RX_ISR(gptimer_handle_t timer, const gptimer_alarm_event_data_t *edata, void *arg) {
199 if (__builtin_expect(bit_counter_RX != 32, 1)) {
200 // Common case: bit_counter_RX is not 32
201 value_RX |= (gpioDirectRead() << bit_counter_RX);
202 bit_counter_RX++;
203 } else {
204 // Less common case: bit_counter_RX is 32
205 cached_gptimer_stop(timer_RX);
206 cached_gpio_isr_handler_add(RX_GPIO_PIN_NUM, RX_gpio_ISR, NULL);
207 cached_ring_buffer_push(ring_buffer_RX, value_RX);
208 reception_complete=true;
209 }
210
211 return true;
212}
